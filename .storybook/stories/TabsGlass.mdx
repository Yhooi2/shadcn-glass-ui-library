## Variants

### Underline (Default)

Active tab with bottom border indicator.

```tsx
<TabsGlass variant="underline" defaultValue="tab1">
  {/* tabs */}
</TabsGlass>
```

### Pill

Active tab with full background highlight.

```tsx
<TabsGlass variant="pill" defaultValue="tab1">
  {/* tabs */}
</TabsGlass>
```

## Examples

### Dashboard Tabs

```tsx
import { BarChart3, Users, Settings } from 'lucide-react';

<TabsGlass defaultValue="analytics">
  <TabsGlass.List>
    <TabsGlass.Trigger value="analytics">
      <BarChart3 className="w-4 h-4 mr-2" />
      Analytics
    </TabsGlass.Trigger>
    <TabsGlass.Trigger value="users">
      <Users className="w-4 h-4 mr-2" />
      Users
    </TabsGlass.Trigger>
    <TabsGlass.Trigger value="settings">
      <Settings className="w-4 h-4 mr-2" />
      Settings
    </TabsGlass.Trigger>
  </TabsGlass.List>

  <TabsGlass.Content value="analytics">
    <StatsGrid />
  </TabsGlass.Content>

  <TabsGlass.Content value="users">
    <UserTable />
  </TabsGlass.Content>

  <TabsGlass.Content value="settings">
    <SettingsForm />
  </TabsGlass.Content>
</TabsGlass>
```

### Controlled Tabs

```tsx
function ControlledTabs() {
  const [activeTab, setActiveTab] = useState('profile');

  return (
    <TabsGlass value={activeTab} onValueChange={setActiveTab}>
      <TabsGlass.List>
        <TabsGlass.Trigger value="profile">Profile</TabsGlass.Trigger>
        <TabsGlass.Trigger value="account">Account</TabsGlass.Trigger>
      </TabsGlass.List>

      <TabsGlass.Content value="profile">
        <ProfileForm />
      </TabsGlass.Content>

      <TabsGlass.Content value="account">
        <AccountSettings />
      </TabsGlass.Content>
    </TabsGlass>
  );
}
```
