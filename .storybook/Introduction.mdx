## Quick Start

```tsx
import { ButtonGlass, ThemeProvider } from 'shadcn-glass-ui';
import 'shadcn-glass-ui/dist/styles.css';

export default function App() {
  return (
    <ThemeProvider defaultTheme="glass">
      <ButtonGlass variant="default">Click me</ButtonGlass>
    </ThemeProvider>
  );
}
```
