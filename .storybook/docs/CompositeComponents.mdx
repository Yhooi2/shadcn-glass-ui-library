## Usage Examples

### MetricCardGlass

```tsx
import { MetricCardGlass } from 'shadcn-glass-ui';

<MetricCardGlass
  label="Trust Score"
  value={87}
  change={+5}
  color="emerald"
  progress={87}
/>

<MetricCardGlass
  label="Response Time"
  value="1.2s"
  change={-0.3}
  color="blue"
/>
```

### ProfileAvatarGlass

```tsx
import { ProfileAvatarGlass } from 'shadcn-glass-ui';

<ProfileAvatarGlass src="/avatar.jpg" name="John Doe" size="xl" status="online" showGlow />;
```

### FlagAlertGlass

```tsx
import { FlagAlertGlass } from 'shadcn-glass-ui';

<FlagAlertGlass
  type="warning"
  title="Attention"
  description="Email verification required"
/>

<FlagAlertGlass
  type="danger"
  title="Error"
  description="Account suspended"
  expanded
/>
```

### AICardGlass

```tsx
import { AICardGlass } from 'shadcn-glass-ui';

<AICardGlass
  title="AI Summary"
  description="Get comprehensive analysis:"
  features={['Code quality assessment', 'Architecture patterns', 'Best practices']}
  ctaText="Generate Report"
  onCtaClick={() => console.log('Generate!')}
/>;
```
