## Quick Start

### ButtonGlass

```tsx
import { ButtonGlass } from 'shadcn-glass-ui';

// Variants (v2.0.0: 100% shadcn/ui compatible)
<ButtonGlass variant="default">Default</ButtonGlass>
<ButtonGlass variant="secondary">Secondary</ButtonGlass>
<ButtonGlass variant="destructive">Destructive</ButtonGlass>
<ButtonGlass variant="ghost">Ghost</ButtonGlass>

// Sizes (v2.0.0: size="md" → size="default")
<ButtonGlass size="sm">Small</ButtonGlass>
<ButtonGlass size="default">Default</ButtonGlass>
<ButtonGlass size="lg">Large</ButtonGlass>
```

### InputGlass

```tsx
import { InputGlass } from 'shadcn-glass-ui';

<InputGlass
  placeholder="Email"
  type="email"
/>

<InputGlass
  label="Password"
  type="password"
  error="Minimum 8 characters"
/>
```

### GlassCard

```tsx
import { GlassCard } from 'shadcn-glass-ui';

<GlassCard variant="glass" intensity="medium">
  <h3>Title</h3>
  <p>Card content</p>
</GlassCard>;
```

## API Patterns

### Compound Components

ModalGlass and TabsGlass support the compound component pattern:

```tsx
// ModalGlass compound API
<ModalGlass.Root open={open} onOpenChange={setOpen}>
  <ModalGlass.Overlay />
  <ModalGlass.Content>
    <ModalGlass.Header>
      <ModalGlass.Title>Title</ModalGlass.Title>
    </ModalGlass.Header>
    <ModalGlass.Body>Content</ModalGlass.Body>
    <ModalGlass.Footer>
      <ButtonGlass>OK</ButtonGlass>
    </ModalGlass.Footer>
  </ModalGlass.Content>
</ModalGlass.Root>
```

### asChild Pattern

ButtonGlass, AvatarGlass, GlassCard support polymorphic rendering:

```tsx
import { Link } from 'react-router-dom';

<ButtonGlass asChild>
  <Link href="/">Home</Link>
</ButtonGlass>;
```
