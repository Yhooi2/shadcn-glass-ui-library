## Using Blocks

### Importing a Block

```tsx
import { ButtonsBlock } from 'shadcn-glass-ui/blocks';

// In your component
function ComponentsDemo() {
  return (
    <div className="p-8">
      <h2>Buttons</h2>
      <ButtonsBlock />
    </div>
  );
}
```

### Customizing a Block

```tsx
import { FormElementsBlock } from 'shadcn-glass-ui/blocks';

<FormElementsBlock
  showLabels
  className="my-custom-class"
  onInputChange={(value) => console.log(value)}
/>;
```

## Creating Custom Blocks

Follow this pattern to create your own blocks:

```tsx
// MyCustomBlock.tsx
import { GlassCard, ButtonGlass } from 'shadcn-glass-ui';

interface MyCustomBlockProps {
  title?: string;
  className?: string;
}

export function MyCustomBlock({ title = 'My Block', className }: MyCustomBlockProps) {
  return (
    <GlassCard className={className}>
      <h3 className="text-lg font-semibold mb-4">{title}</h3>

      <div className="flex flex-wrap gap-4">
        {/* Your components */}
        <ButtonGlass variant="primary">Action 1</ButtonGlass>
        <ButtonGlass variant="secondary">Action 2</ButtonGlass>
      </div>
    </GlassCard>
  );
}
```
