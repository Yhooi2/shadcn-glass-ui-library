## Usage Examples

### StatusIndicatorGlass

```tsx
import { StatusIndicatorGlass } from 'shadcn-glass-ui';

// User statuses
<StatusIndicatorGlass status="online" />
<StatusIndicatorGlass status="offline" />
<StatusIndicatorGlass status="busy" />
<StatusIndicatorGlass status="away" />

// With sizes
<StatusIndicatorGlass status="online" size="sm" />
<StatusIndicatorGlass status="online" size="md" />
<StatusIndicatorGlass status="online" size="lg" />
```

### SegmentedControlGlass

```tsx
import { SegmentedControlGlass } from 'shadcn-glass-ui';

const options = [
  { value: 'day', label: 'Day' },
  { value: 'week', label: 'Week' },
  { value: 'month', label: 'Month' },
];

<SegmentedControlGlass options={options} value={selected} onChange={setSelected} />;
```

### RainbowProgressGlass

```tsx
import { RainbowProgressGlass } from 'shadcn-glass-ui';

<RainbowProgressGlass value={75} />
<RainbowProgressGlass value={100} animated />
```

### LanguageBarGlass

```tsx
import { LanguageBarGlass } from 'shadcn-glass-ui';

const languages = [
  { name: 'English', level: 'native' },
  { name: 'Spanish', level: 'fluent' },
  { name: 'German', level: 'intermediate' },
];

<LanguageBarGlass languages={languages} showLegend />;
```
